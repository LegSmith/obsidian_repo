```java
package jpabook.jpashop.repository;

import jakarta.persistence.EntityManager;
import jakarta.persistence.PersistenceContext;
import jpabook.jpashop.domain.Member;
import org.springframework.stereotype.Repository;

import java.util.List;

@Repository //스프링 빈에 등록
@RequiredArgsConstructor 
public class MemberRepository {

    //@PersistenceContext //spring이 엔티티매니저를 만들어 주입(injection)해준다.
    private final EntityManager em;

    public void save(Member member){
        em.persist(member);
    }

    public Member findOne(Long id){
        return em.find(Member.class,id);
    }

    public List<Member> findAll(){
        return em.createQuery("select m from Member m", Member.class).getResultList();
    }

    public List<Member> findByName(String name){
        return em.createQuery("select m from Member m where m.name=:name",Member.class)
                .setParameter("name",name) // name으로 바인딩이 된다.
                .getResultList();
    }
}
```

- **@Repository** : ==_**스프링 빈에 등록**_==시키는 어노테이션
- **@PersistenceContext** : JPA에 엔티티매니저를 ==_**스프링이 생성한 엔티티메니저에 주입(injection)**_==시켜준다.
- **persist()** 메서드는 객체를 영속성 컨텍스트에 주입하는 메서드
- **find()**메서드는 단건조회 시에 사용하는 메서드
- **createQuery()**는 JPQL 명령어를 입력하고 객체의 타입을 파라미터로 받는 메서드이다
- **getResultList()**는 _**==쿼리문으로 나온 결과를 리스트형식으로 변환==**_해주는 메서드
- **setParameter()**메서드는 _**==JPQL의 파라미터를 바인딩==**_해주는 메서드이다.

  

## 회원 서비스 구현

==_**JPA에 모든 데이터 변경이나 로직들은 트랜잭션(Transaction)안에서 실행되어야 한다!!**_==

```java
package jpabook.jpashop.service;

import jpabook.jpashop.domain.Member;
import jpabook.jpashop.repository.MemberRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;

@Service
@RequiredArgsConstructor //final을 가지고 있는 필드만 생성자 주입을 시켜준다.
@Transactional(readOnly = true)
public class MemberService {

    private final MemberRepository memberRepository;

    /**
     * 회원가입
     */
    @Transactional
    public Long join(Member member){
        //같은 이름 중복체크
        validateDuplicateMember(member);
        //중복이 아니면 아래로직 실행
        memberRepository.save(member);
        return member.getId();
    }

    private void validateDuplicateMember(Member member) {
        //여기서 예외발생하면 터뜨린다.
        List<Member> findMembers = memberRepository.findByName(member.getName());
        if(!findMembers.isEmpty()){ //중복이라면
            throw new IllegalStateException("이미 존재하 회원입니다.");
        }
    }// 실무에서는 멀티 쓰레드의 문제등이 있기때문에 이방법을 권장하진 않는다.
     // 실제 db에 name컬럼을 유니크 제약조건을 거는것을 추천한다.

    /**
     * 회원 전체 조회
     */
    public List<Member> findMembers(){
        return memberRepository.findAll();
    }

    /**
     * 회원 한건 조회
     */
    public Member findOne(Long memberId){
        return memberRepository.findOne(memberId);
    }
}
```

- **@Transactional(readOnly = true)** : 조회하는 메서드에 걸어주면 ==_**DB성능을 최적화**_== 시킬 수 있다.
- **@Transactional** : ==_**데이터 변경등의 로직은 하나의 트랜잭션안에서 일어나야**_== 하기 때문에 걸어준다.
    
    **조회만 하는 메서드가 많기 때문에 클래스 전체에 @Transactional(readOnly = true)를 걸어주고 조회가 아닌 메서드 위에 @Transactional 을 붙혀준다.**
    
- **@Autowired** : 스프링 빈에 등록되어 있는 MemberRepository를 MemberService에 주입시켜준다
- **@RequiredArgsConstructor** : 스프링 부트에서는 final이 붙은 필드를 생성자 주입을 해준다.( 이방법을 추천)

  

# 테스트 코드

- `@RunWith(SpringRunner.class)` : _**jUnit실행시에 스프링이랑 엮어서 실행**_시키는 코드
- `@SpringBootTest` : _**Springboot를 띄어놓고 테스트를 실행**_( 없다면 @Autowired 같은 어노테이션이 적용되지 않는다 ! )
- `@Transactional` : 트랜잭션을 걸고 _**테스트가 끝나면 모두 rollback**_을 시킨다(특정 메서드에대해 ==_**commit을 하고 싶으면 @Rollback(false)**_==를 적으면 된다)

## 회원 가입 테스트 코드

```java
package jpabook.jpashop.service;

import jpabook.jpashop.domain.Member;
import jpabook.jpashop.repository.MemberRepository;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.context.junit4.SpringRunner;
import org.springframework.transaction.annotation.Transactional;

import static org.junit.Assert.*;

@RunWith(SpringRunner.class)
@SpringBootTest
@Transactional
public class MemberServiceTest {

    @Autowired MemberService memberService;
    @Autowired MemberRepository memberRepository;

    @Test
    public void 회원가입() throws Exception {
        //given
        Member member = new Member();
        member.setName("Ronaldo");

        //when
        Long savedId = memberService.join(member);

        //then
        assertEquals(member,memberRepository.findOne(savedId));
    }

}
```

- 회원가입 테스트 실행시 쿼리문을 보면 insert문이 없다. 왜냐하면 persists(member)시에는 영속성 컨텍스트에 들어가지 바로 DB에 쿼리문을 보내지 않는다(지연로딩) .
- 크리고 테스트 클래스에 @Transactional은 기본적으로 테스트를하고 rollback을 하기 때문에 insert쿼리문이 나가지 않는다.
- select문이 나가는 이유는 join메서드에 이름중복메서드에서 쿼리문을 실행시키기 때문에 select가 나가는것이다.
    
    ![[Untitled 14.png|Untitled 14.png]]
    
- 직접 데이터가 들어가는것을 보고싶다면 테스트 메서드에 @Rollback(false)를 붙혀준다.
    
    ```java
    package jpabook.jpashop.service;
    
    import jpabook.jpashop.domain.Member;
    import jpabook.jpashop.repository.MemberRepository;
    import org.junit.Test;
    import org.junit.runner.RunWith;
    import org.springframework.beans.factory.annotation.Autowired;
    import org.springframework.boot.test.context.SpringBootTest;
    import org.springframework.test.context.junit4.SpringRunner;
    import org.springframework.transaction.annotation.Transactional;
    
    import static org.junit.Assert.*;
    
    @RunWith(SpringRunner.class)
    @SpringBootTest
    @Transactional
    public class MemberServiceTest {
    
        @Autowired MemberService memberService;
        @Autowired MemberRepository memberRepository;
    
        @Test
    		@Rollback(false)
        public void 회원가입() throws Exception {
            //given
            Member member = new Member();
            member.setName("Ronaldo");
    
            //when
            Long savedId = memberService.join(member);
    
            //then
            assertEquals(member,memberRepository.findOne(savedId));
        }
    
    }
    ```

## 중복회원예외 테스트 코드

```java
package jpabook.jpashop.service;

import jpabook.jpashop.domain.Member;
import jpabook.jpashop.repository.MemberRepository;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.annotation.Rollback;
import org.springframework.test.context.junit4.SpringRunner;
import org.springframework.transaction.annotation.Transactional;

import static org.junit.Assert.*;

@RunWith(SpringRunner.class)
@SpringBootTest
@Transactional
public class MemberServiceTest {

    @Autowired MemberService memberService;
    @Autowired MemberRepository memberRepository;

    @Test
    public void 중복_회원_예외() throws Exception{
        //given
        Member member1 = new Member();
        member1.setName("Ronaldo");

        Member member2 = new Member();
        member2.setName("Ronaldo");
        //when
        memberService.join(member1);
        try {
            memberService.join(member2); //여기서 예외가 발생해야함 !!
        }catch (IllegalStateException e){
            return;
        }
        //then
        fail("예외가 발생해야 한다");
    }

}
```

- 위 코드에서 member1과 똑같은 이름을 가진 member2를 조인하면 IllegalStateException이 터지게 로직을 짯다. 그래서 try-catch문을 통해서 IllegalStateException이 발생하면 return을 하여 테스트를 성공시킨다.
- fail()메서드는 Assert에 있는 메서드로 실패메세지를 띄우는 것이다.
- 위 코드를 더 깔끔하게 짠다면 아래와 같이 짤 수 있다.

```java
@Test(expected = IllegalStateException.class)
    public void 중복_회원_예외() throws Exception{
        //given
        Member member1 = new Member();
        member1.setName("Ronaldo");

        Member member2 = new Member();
        member2.setName("Ronaldo");
        //when
        memberService.join(member1);
        memberService.join(member2); //여기서 예외가 발생해야함 !!
        //then
        fail("예외가 발생해야 한다");
    }
```
